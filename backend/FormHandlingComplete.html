4 Handling Input Change (onChange)

üìå Explanation:

Whenever the user types in an input field, onChange is triggered.

I update the correct field using event.target.name and event.target.value.

How it works internally:

  • e.target.name ‚Üí tells which input
  • e.target.value ‚Üí tells new value
const handleOnChange = (e) => {
  // e = event object
  // e.target = the input element that triggered the event
  // e.target.name = name attribute of input (e.g., "email")
  // e.target.value = current value typed by user
  
  setFormData((prev) => ({
    ...prev,  // Keep all existing values (spread operator)
    [e.target.name]: e.target.value  // Update only the changed field
  }));
};

// Example flow:
// User types "John" in name field
// e.target.name = "name"
// e.target.value = "John"
// Result: { name: "John", email: "", password: "" }
üîç Detailed Breakdown: 
Before: formData = { name: "", email: "test@example.com", password: "123" }
User types "John" in name field

setFormData((prev) => {
  // prev = { name: "", email: "test@example.com", password: "123" }
  // ...prev spreads it: name: "", email: "test@example.com", password: "123"
  // [e.target.name]: e.target.value adds: name: "John"
  // Final: { name: "John", email: "test@example.com", password: "123" }
});

6 Validating Inputs (Before API Call)

‚úÖ Simple Validation:

üîç Advanced Validation with Regex:

const handleOnSubmit = (e) => {
  e.preventDefault();  // Prevents page reload
  
  // Email Regex Pattern
  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  
  // Phone Regex Pattern (10 digits)
  const phoneRegex = /^[0-9]{10}$/;
  
  // Password Regex (min 6 chars, 1 uppercase, 1 number, 1 special char)
  const passwordRegex = /^(?=.*[A-Z])(?=.*\d)(?=.*[@$!%*?&])[A-Za-z\d@$!%*?&]{6,}$/;
  
  // Validation checks
  if (!formData.name || formData.name.trim() === "") {
    return toast.error("Name is required");
  }
  
  if (!emailRegex.test(formData.email)) {
    return toast.error("Invalid email format");
  }
  
  if (!phoneRegex.test(formData.phone)) {
    return toast.error("Phone must be 10 digits");
  }
  
  if (!passwordRegex.test(formData.password)) {
    return toast.error("Password must have 6+ chars, 1 uppercase, 1 number, 1 special char");
  }
  
  if (formData.password !== formData.confirmPassword) {
    return toast.error("Passwords do not match");
  }
  
  // If all validations pass, send data
  sendDataToBackend();
};
üéØ Regex Explanation:
  • Email: /^[^\s@]+@[^\s@]+\.[^\s@]+$/ - Ensures format like user@example.com
  • Phone: /^[0-9]{10}$/ - Exactly 10 digits
  • Password: Must contain uppercase, number, special character, minimum 6 characters

9 Sending Data to Backend - Axios vs Fetch

üìå Explanation:

After validation, we send data to the backend using either Axios or Fetch API.

Option 1: Using Axios (Simpler)

import axios from "axios";

const sendDataToBackend = async () => {
  try {
    const response = await axios.post("http://localhost:5000/api/signup", {
      name: formData.name,
      email: formData.email,
      password: formData.password,
      phone: formData.phone
    });
    
    toast.success("Signup successful!");
    console.log(response.data);
  } catch (error) {
    toast.error(error.response?.data?.message || "Signup failed");
  }
};
‚úÖ Axios Benefits:
  • Automatically converts data to JSON - no need for JSON.stringify()
  • Automatically parses response - no need for response.json()
  • Better error handling with error.response
  • Cleaner and more concise code

Option 2: Using Fetch (Built-in)

const sendDataToBackend = async () => {
  try {
    const response = await fetch("http://localhost:5000/api/signup", {
      method: "POST",
      headers: {
        "Content-Type": "application/json"
      },
      body: JSON.stringify({
        name: formData.name,
        email: formData.email,
        password: formData.password,
        phone: formData.phone
      })
    });
    
    const data = await response.json();
    
    if (response.ok) {
      toast.success("Signup successful!");
      console.log(data);
    } else {
      toast.error(data.message || "Signup failed");
    }
  } catch (error) {
    toast.error("Network error occurred");
  }
};
‚ö†Ô∏è Fetch Requirements:
  • Must specify method: "POST"
  • Must set Content-Type: application/json header
  • Must use JSON.stringify() to convert data to JSON string
  • Must manually call response.json() to parse response
  • Must manually check response.ok for success

11 Backend - API Endpoint with Validation

üìå Explanation:

We create an API endpoint that:

  • Extracts form data from request body
  • Validates the data
  • Uses try-catch for error handling
  • Saves data to database
  • Returns appropriate response
// signupController.js
const User = require("../models/User");
const bcrypt = require("bcrypt");

exports.signup = async (req, res) => {
  try {
    // Step 1: Extract form data from request body
    const {
      name,
      gender,
      email,
      password,
      phone,
      department,
      year,
      college,
      role
    } = req.body;
    
    // Step 2: Validate data
    if (!name || !email || !password || !phone) {
      return res.status(400).json({
        success: false,
        message: "All fields are required"
      });
    }
    
    // Email validation
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (!emailRegex.test(email)) {
      return res.status(400).json({
        success: false,
        message: "Invalid email format"
      });
    }
    
    // Password length validation
    if (password.length < 6) {
      return res.status(400).json({
        success: false,
        message: "Password must be at least 6 characters"
      });
    }
    
    // Step 3: Check if user already exists
    const existingUser = await User.findOne({ email });
    if (existingUser) {
      return res.status(409).json({
        success: false,
        message: "User already exists with this email"
      });
    }
    
    // Step 4: Hash password for security
    const hashedPassword = await bcrypt.hash(password, 10);
    
    // Step 5: Create new user using "new" keyword and save
    const newUser = new User({
      name,
      gender,
      email,
      password: hashedPassword,
      phone,
      department,
      year,
      college,
      role
    });
    
    // Step 6: Save to database
    await newUser.save();
    
    // Step 7: Send success response with 200 status
    return res.status(200).json({
      success: true,
      message: "User registered successfully",
      data: {
        id: newUser._id,
        name: newUser.name,
        email: newUser.email,
        role: newUser.role
      }
    });
    
  } catch (error) {
    // Step 8: Handle errors
    console.error("Signup Error:", error);
    return res.status(500).json({
      success: false,
      message: "Internal server error. Please try again later",
      error: error.message
    });
  }
};
